## Object
#### Property
An Object is a collection of properties.
Three kinds of properties:
Named property 

var a = {
  b: 100
};

var descriptor = Object.getOwnPropertyDescriptor(a, "b");

console.log(descriptor);
// {value: 100, writable: true, enumerable: true, configurable: true}
Accessor property 

var x = 10;
var a = Object.create({}, {
  b: {
    get: function() {
      return x;
    },
    set: function(v) {
      x = v;
    }
  }
});
var descriptor = Object.getOwnPropertyDescriptor(a, "b");
console.log(descriptor);

console.log(a.b); // 10
a.b = 20;
console.log(x);   // 20
Internal property

var a = {

console.log(b.__proto__ === a); // true

this 

function printThis() {
  console.log(this);
}

printThis(); // window
var a = {
  p: printThis
};

a.p(); // a

printThis.call({x: 10}); // {x: 10}
printThis.apply({y: 10}) // {y: 10}

var printThis = function() {
  console.log(this)
}.bind({z: 10});

printThis(); // {z: 10}
a.p = printThis;
a.p() // {z: 10}
printThis.call(window); // {z: 10}
[[scope]] of function 

module.exports = once;

function once(fn) {
  var ran = false;
  var result;
  return function() {
    if (!ran) {
      ran = true;
      result = fn.apply(this, arguments);
    }
    return result;
  }
}
						

var f = function factorial(n) {
  if (n == 1) return n;
  return n * factorial(n - 1);
}
